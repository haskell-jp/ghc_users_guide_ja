data MenuItemInfo
 = MenuItemInfo  {
      menuItemType    :: MenuFlag,
      menuItemState   :: MenuState,
      menuItemID      :: UINT,
      menuItemSubMenu :: HMENU,
      menuItemBitmapChecked :: HBITMAP,
      menuItemBitmapUnchecked :: HBITMAP,
      menuItemData    :: DWORD,
      menuItemTypeData :: String
   }

-- Don't make this an instance of Storable, because poke isn't what we want.

peekMenuItemInfo :: Ptr MenuItemInfo -> IO MenuItemInfo
peekMenuItemInfo p = do
  itemType <- (\hsc_ptr -> peekByteOff hsc_ptr 8) p
{-# LINE 286 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  itemState <- (\hsc_ptr -> peekByteOff hsc_ptr 12) p
{-# LINE 287 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  itemID <- (\hsc_ptr -> peekByteOff hsc_ptr 16) p
{-# LINE 288 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  itemSubMenu <- (\hsc_ptr -> peekByteOff hsc_ptr 24) p
{-# LINE 289 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  itemBitmapChecked <- (\hsc_ptr -> peekByteOff hsc_ptr 32) p
{-# LINE 290 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  itemBitmapUnchecked <- (\hsc_ptr -> peekByteOff hsc_ptr 40) p
{-# LINE 291 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  itemData <- (\hsc_ptr -> peekByteOff hsc_ptr 48) p
{-# LINE 292 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  nchars <- (\hsc_ptr -> peekByteOff hsc_ptr 64) p
{-# LINE 293 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  c_str <- (\hsc_ptr -> peekByteOff hsc_ptr 56) p
{-# LINE 294 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  itemTypeData <- peekTStringLen (c_str, fromIntegral (nchars::UINT))
  return MenuItemInfo
    { menuItemType = itemType
    , menuItemState = itemState
    , menuItemID = itemID
    , menuItemSubMenu = itemSubMenu
    , menuItemBitmapChecked = itemBitmapChecked
    , menuItemBitmapUnchecked = itemBitmapUnchecked
    , menuItemData = itemData
    , menuItemTypeData = itemTypeData
    }

allocaMenuItemInfo :: (Ptr MenuItemInfo -> IO a) -> IO a
allocaMenuItemInfo f =
  let size = (80) in
{-# LINE 309 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  allocaBytes size $ \ p -> do
  (\hsc_ptr -> pokeByteOff hsc_ptr 0) p (fromIntegral size::DWORD)
{-# LINE 311 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  f p

withMenuItemInfo :: MenuItemInfo -> (Ptr MenuItemInfo -> IO a) -> IO a
withMenuItemInfo info f =
  allocaMenuItemInfo $ \ p ->
  withTStringLen (menuItemTypeData info) $ \ (c_str, nchars) -> do
  (\hsc_ptr -> pokeByteOff hsc_ptr 8) p (menuItemType info)
{-# LINE 318 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  (\hsc_ptr -> pokeByteOff hsc_ptr 12) p (menuItemState info)
{-# LINE 319 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  (\hsc_ptr -> pokeByteOff hsc_ptr 16) p (menuItemID info)
{-# LINE 320 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  (\hsc_ptr -> pokeByteOff hsc_ptr 24) p (menuItemSubMenu info)
{-# LINE 321 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  (\hsc_ptr -> pokeByteOff hsc_ptr 32) p (menuItemBitmapChecked info)
{-# LINE 322 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  (\hsc_ptr -> pokeByteOff hsc_ptr 40) p (menuItemBitmapUnchecked info)
{-# LINE 323 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  (\hsc_ptr -> pokeByteOff hsc_ptr 48) p c_str
{-# LINE 324 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  (\hsc_ptr -> pokeByteOff hsc_ptr 64) p (fromIntegral nchars::UINT)
{-# LINE 325 "libraries\\Win32\\Graphics\\Win32\\Menu.hsc" #-}
  f p
